def queryize_params(hash)
  out = {}
  hash.each do |k, v|
    k = 'Attribute' if k == :attributes    # Allow plural for readability
    key = camelkey(k)
    case v
    when Hash
      out.merge! flatten_subhash(key, v)
    when Array
      out.merge! flatten_subarray(key, v)
    else
      out[key] = v
    end
  end
  out
end
